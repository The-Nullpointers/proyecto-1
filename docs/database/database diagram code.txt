TABLE USERS{
  USERNAME VARCHAR2(10) [PRIMARY KEY]
  PASSWORD VARCHAR2(32)
  AVATARS INT
}

TABLE AVATARS{
  ID_AVATAR INT [PRIMARY KEY]
  USERNAME VARCHAR2(10)
  ID_COLOR INT
  ID_HEAD INT
  ID_TORSO INT
  ID_ACCESSORY INT
}

TABLE COLORS{
  ID_COLOR INT [PRIMARY KEY]
  COLOR_NAME VARCHAR2(32)
}

TABLE HEADS{
  ID_HEAD INT [PRIMARY KEY]
  HEAD_NAME VARCHAR2(32)
}

TABLE TORSOS{
  ID_TORSO INT [PRIMARY KEY]
  TORSO_NAME VARCHAR2(32)
}

TABLE ACCESSORIES{
  ID_ACCESSORY INT [PRIMARY KEY]
  ACCESSORY_NAME VARCHAR2(32)
}

REF: AVATARS.USERNAME > USERS.USERNAME
REF: AVATARS.ID_COLOR > COLORS.ID_COLOR
REF: AVATARS.ID_HEAD > HEADS.ID_HEAD
REF: AVATARS.ID_TORSO > TORSOS.ID_TORSO
REF: AVATARS.ID_ACCESSORY > ACCESSORIES.ID_ACCESSORY
